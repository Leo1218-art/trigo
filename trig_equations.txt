        function getRandomQuestions() {
            const shuffled = questionBank.sort(() => 0.5 - Math.random());
            return shuffled.slice(0, 10);
        }

        function renderQuiz() {
            const quizDiv = document.getElementById('quiz');
            quizDiv.innerHTML = '';
            const questions = getRandomQuestions();
            questions.forEach((q, index) => {
                const questionDiv = document.createElement('div');
                questionDiv.className = 'question';
                questionDiv.innerHTML = `
                    <p><strong>Question ${index + 1}:</strong> ${q.question}</p>
                    <button onclick="showSolution(${index})">Show Solution</button>
                    <div id="solution${index}" class="solution">
                        <p><strong>Answer:</strong> ${q.answer}</p>
                        <p><strong>Solution:</strong></p>
                        <ul>
                            ${q.solution.map(step => `<li>${step}</li>`).join('')}
                        </ul>
                    </div>
                `;
                quizDiv.appendChild(questionDiv);
            });
            MathJax.typeset();
        }

        function showSolution(index) {
            const solutionDiv = document.getElementById(`solution${index}`);
            solutionDiv.style.display = solutionDiv.style.display === 'block' ? 'none' : 'block';
            MathJax.typeset();
        }

        document.getElementById('tryAgain').addEventListener('click', renderQuiz);

        // Initial render
        renderQuiz();
